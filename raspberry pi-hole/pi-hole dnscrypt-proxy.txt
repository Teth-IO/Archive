#DNSCrypt
cd /opt
wget https://github.com/DNSCrypt/dnscrypt-proxy/releases/download/2.1.1/dnscrypt-proxy-linux_arm64-2.1.1.tar.gz
(voir https://github.com/DNSCrypt/dnscrypt-proxy/releases pour la dernière release)
tar -xzf dnscrypt-proxy-linux_arm64-2.1.1.tar.gz
mv linux-arm dnscrypt-proxy
cd dnscrypt-proxy
cp example-dnscrypt-proxy.toml dnscrypt-proxy.toml
vim dnscrypt-proxy.toml
	listen_addresses = ['(localhost ipv6):5300'] (comme 53 est deja pris par pi-hole)
	## on configure DNSCrypt + DNSSEC + non-logging et non-blocking
./dnscrypt-proxy -service install
systemctl start dnscrypt-proxy
systemctl enable dnscrypt-proxy

-> la suite https://github.com/pi-hole/pi-hole/wiki/DNSCrypt-2.0
